public static function getTimeago($timestamp) { $estimate_time = time() - $timestamp; if ($estimate_time < 1) { return 'a second ago'; } $condition = [ 12 * 30 * 24 * 60 * 60 => 'year', 30 * 24 * 60 * 60 => 'month', 24 * 60 * 60 => 'day', 60 * 60 => 'hour', 60 => 'minute', 1 => 'second' ]; foreach ($condition as $secs => $str) { $d = $estimate_time / $secs; if ($d >= 1) { $r = round($d); return $r . ' ' . $str . ( $r > 1 ? 's' : '' ) . ' ago'; } } }
